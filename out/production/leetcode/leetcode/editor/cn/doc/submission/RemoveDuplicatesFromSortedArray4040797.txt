class Solution:
    def removeDuplicates(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        n = len(nums)
        sum = 1
        count = 1
        if n == 0: return 0
        else:
            for i in range(n - 1):
                if nums[i] != nums[i +1]:
                    sum += 1
                    nums[count] = nums[i+1]
                    count += 1
            return sum
